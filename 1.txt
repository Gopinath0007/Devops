<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <system.webServer>
        <rewrite>
            <!-- Inbound rules for Application Servers -->
            <rules>
                <!-- Inbound rule for Application Server 1 -->
                <rule name="ReverseProxyInboundRule1" stopProcessing="true">
                    <match url="(.*)" />
                    <action type="Rewrite" url="http://10.23.32.57:8081/{R:1}" />
                    <!-- No conditions needed here -->
                </rule>

                <!-- Inbound rule for Application Server 2 (failover) -->
                <rule name="ReverseProxyInboundRule2" stopProcessing="true">
                    <match url="(.*)" />
                    <action type="Rewrite" url="http://10.23.32.55:8081/{R:1}" />
                    <!-- Conditions to check if the response status is not 200 -->
                    <conditions>
                        <add input="{RESPONSE_STATUS}" pattern="^200$" negate="true" />
                    </conditions>
                </rule>
            </rules>

            <!-- Outbound rules for rewriting links in the response -->
            <outboundRules>
                <!-- Outbound rule for Application Server 1 -->
                <rule name="ReverseProxyOutboundRule1" preCondition="ResponseIsHtml1">
                    <match filterByTags="A, Form, Img" pattern="^http(s)?://10.23.32.57:8081/(.*)" />
                    <action type="Rewrite" value="http{R:1}://eba-uat.addof.gov.ae/{R:2}" />
                </rule>

                <!-- Outbound rule for Application Server 2 -->
                <rule name="ReverseProxyOutboundRule2" preCondition="ResponseIsHtml2">
                    <match filterByTags="A, Form, Img" pattern="^http(s)?://10.23.32.55:8081/(.*)" />
                    <action type="Rewrite" value="http{R:1}://eba-uat.addof.gov.ae/{R:2}" />
                </rule>
            </outboundRules>

            <!-- Preconditions for detecting HTML responses -->
            <preConditions>
                <!-- Precondition for detecting HTML responses from Application Server 1 -->
                <preCondition name="ResponseIsHtml1">
                    <add input="{RESPONSE_CONTENT_TYPE}" pattern="^text/html" />
                </preCondition>

                <!-- Precondition for detecting HTML responses from Application Server 2 -->
                <preCondition name="ResponseIsHtml2">
                    <add input="{RESPONSE_CONTENT_TYPE}" pattern="^text/html" />
                </preCondition>
            </preConditions>
        </rewrite>
    </system.webServer>
</configuration>
